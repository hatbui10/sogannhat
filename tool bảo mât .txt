<script>
    let currentShape = '';
    const shapes = [
        '<circle cx="12" cy="12" r="9"/>',
        '<rect x="4" y="4" width="16" height="16"/>',
        '<polygon points="12,2 22,20 2,20"/>',
        '<ellipse cx="12" cy="12" rx="9" ry="5"/>',
        '<path d="M12 2 L15 9 L22 9 L17 14 L19 21 L12 17 L5 21 L7 14 L2 9 L9 9 Z"/>', // Star
        '<rect x="6" y="6" width="12" height="12" rx="2"/>',
        '<circle cx="12" cy="12" r="4"/>',
        '<path d="M3 12 h18 M12 3 v18"/>', // Cross
        '<path d="M5 5 L19 19 M19 5 L5 19"/>', // X
        '<path d="M12 21 a9 9 0 1 0 0 -18 v18"/>' // Semi-circle
    ];

    // Khởi tạo 10 hình vẽ
    const grid = document.getElementById('shapeGrid');
    shapes.forEach((s, i) => {
        const div = document.createElement('div');
        div.className = 'shape-item';
        div.innerHTML = `<svg viewBox="0 0 24 24">${s}</svg>`;
        div.onclick = () => {
            currentShape = i;
            document.querySelectorAll('.shape-item').forEach(el => el.classList.remove('selected'));
            div.classList.add('selected');
        };
        grid.appendChild(div);
    });

    function resetForm() {
        document.getElementById('fileInput').value = '';
        document.getElementById('answer').value = '';
        document.getElementById('digits').value = '';
        document.getElementById('specials').value = '';
        document.getElementById('question').selectedIndex = 0;
        currentShape = '';
        document.querySelectorAll('.shape-item').forEach(el => el.classList.remove('selected'));
    }

    async function handleProcess(mode) {
        const file = document.getElementById('fileInput').files[0];
        const ans = document.getElementById('answer').value;
        const dig = document.getElementById('digits').value;
        const spec = document.getElementById('specials').value;

        if (!file || !ans || dig.length < 1 || currentShape === '') {
            alert("Thiếu thông tin! Cần File + Trả lời + Số + Chọn hình.");
            return;
        }

        const key = ans + dig + spec + currentShape;
        const reader = new FileReader();

        reader.onload = function(e) {
            const data = e.target.result;
            try {
                if (mode === 'encrypt') {
                    const encrypted = CryptoJS.AES.encrypt(data, key).toString();
                    const blob = new Blob([encrypted], {type: 'text/plain'});
                    const a = document.createElement('a');
                    const rand = Math.floor(1000 + Math.random() * 9000);
                    a.href = URL.createObjectURL(blob);
                    a.download = `SECURE_${rand}.txt`;
                    a.click();
                    
                    // Reset ngay sau khi mã hóa
                    resetForm();
                    alert("Đã mã hóa và tải file! Form đã được dọn dẹp.");
                } else {
                    const decrypted = CryptoJS.AES.decrypt(data, key);
                    const original = decrypted.toString(CryptoJS.enc.Utf8);
                    if (!original) throw new Error();
                    document.getElementById('output').value = original;
                }
            } catch (err) {
                alert("Lỗi: Sai thông tin bảo mật hoặc file không đúng định dạng!");
            }
        };

        reader.readAsText(file);
    }
</script>
