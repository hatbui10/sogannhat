<script>
    const fileInput = document.getElementById('fileInput'),
          rowLengthInput = document.getElementById('rowLength'),
          stepKInput = document.getElementById('stepK'),
          totalCharsInput = document.getElementById('totalChars'),
          statusBar = document.getElementById('status-bar'),
          resultGrid = document.getElementById('result-grid'),
          resultContainer = document.getElementById('result-container'),
          headerWrapper = document.getElementById('header-wrapper'),
          listItems = document.getElementById('list-items'),
          searchResults = document.getElementById('search-results'),
          inputG = document.getElementById('inputG'),
          outputP = document.getElementById('outputP');

    let currentNumbers = [];
    let highlightR = -1;
    let highlightR1 = -1;

    resultContainer.addEventListener('scroll', () => {
        headerWrapper.scrollLeft = resultContainer.scrollLeft;
    });

    function showStatus(text, type) {
        statusBar.textContent = text; statusBar.className = 'status-' + type;
        statusBar.style.display = 'block';
    }

    function clearData() {
        if(confirm("Xóa toàn bộ dữ liệu đã lưu?")) { 
            localStorage.clear(); 
            location.reload(); 
        }
    }

    window.onload = function() {
        rowLengthInput.value = localStorage.getItem('userX') || 2;
        stepKInput.value = localStorage.getItem('userK') || 1;
        const savedNums = localStorage.getItem('userNums');
        if (savedNums) { 
            currentNumbers = JSON.parse(savedNums); 
            totalCharsInput.value = currentNumbers.length;
        }
        const savedMode = localStorage.getItem('userMode') || 'stepK';
        updateActiveButton(savedMode);
        if (currentNumbers.length > 0) refresh();
    };

    fileInput.onchange = function(e) {
        const file = e.target.files[0];
        if (!file) return;
        const reader = new FileReader();
        reader.onload = function(event) {
            currentNumbers = event.target.result.replace(/[^0-9]/g, '').split('').map(Number);
            totalCharsInput.value = currentNumbers.length;
            localStorage.setItem('userNums', JSON.stringify(currentNumbers));
            refresh();
        };
        reader.readAsText(file);
    };

    function updateActiveButton(mode) {
        document.querySelectorAll('.main-btn').forEach(b => b.classList.remove('active-mode'));
        const activeBtn = document.getElementById('btn-' + mode);
        if (activeBtn) activeBtn.classList.add('active-mode');
    }

    function setMode(mode) {
        highlightR = -1; highlightR1 = -1; outputP.value = '';
        localStorage.setItem('userMode', mode);
        updateActiveButton(mode);
        refresh();
    }

    function refresh() {
        const x = parseInt(rowLengthInput.value) || 2;
        const k = parseInt(stepKInput.value) || 1;
        localStorage.setItem('userX', x); 
        localStorage.setItem('userK', k);
        updateHeader(x); 
        render(x, k);
    }

    function updateHeader(x) {
        const grid = document.getElementById('header-grid');
        grid.style.gridTemplateColumns = `var(--stt-width) var(--spacer-width) repeat(${x}, var(--cell-size))`;
        grid.innerHTML = '<div class="row-stt" style="background:transparent; border:1px dashed #333"></div><div style="width:var(--spacer-width)"></div>';
        for (let i = 0; i < x; i++) {
            const box = document.createElement('div');
            box.className = `index-box idx-${i % 10}`;
            box.textContent = i % 10;
            grid.appendChild(box);
        }
    }

    function autoSearchAll() {
        if (!currentNumbers.length) return alert("Vui lòng chọn file dữ liệu trước!");
        listItems.innerHTML = ''; 
        searchResults.style.display = 'block';
        let found = 0;
        for (let x = 2; x <= 9; x++) {
            for (let k = 1; k <= 30; k++) {
                const u = checkU(currentNumbers, x, k);
                if (u >= 0 && u <= 1) {
                    const item = document.createElement('div');
                    item.className = 'result-item';
                    item.innerHTML = `X:<b>${x}</b> | K:<b>${k}</b> (u=${u})`;
                    item.onclick = () => { 
                        rowLengthInput.value = x; 
                        stepKInput.value = k; 
                        refresh(); 
                    };
                    listItems.appendChild(item); found++;
                }
            }
        }
        showStatus(`Tìm thấy ${found} bộ số có u ≤ 1!`, "info");
    }

    function checkU(data, x, k) {
        let u = 0; const rows = [];
        for (let i = 0; i < data.length; i += x) rows.push(data.slice(i, i + x));
        for (let r = 0; r < rows.length - k; r++) {
            if (rows[r].length === x && rows[r+k].length === x) {
                let isRev = true;
                for (let c = 0; c < x; c++) { 
                    if ((rows[r][c] >= 5) === (rows[r+k][c] >= 5)) { isRev = false; break; } 
                }
                if (isRev) u++;
            }
        }
        return u;
    }

    function searchReverseG() {
        const rIdx = parseInt(inputG.value) - 1;
        const x = parseInt(rowLengthInput.value) || 2;
        if (!currentNumbers.length) return alert("Chưa có dữ liệu!");
        
        const rows = [];
        for (let i = 0; i < currentNumbers.length; i += x) rows.push(currentNumbers.slice(i, i + x));
        
        if (rIdx < 0 || rIdx >= rows.length) return alert("STT hàng không hợp lệ!");
        
        const targetRow = rows[rIdx];
        highlightR = rIdx; 
        highlightR1 = -1;
        outputP.value = '';

        for (let i = rIdx + 1; i < rows.length; i++) {
            if (rows[i].length === x) {
                let isAllReverse = true;
                for (let c = 0; c < x; c++) {
                    if ((targetRow[c] >= 5) === (rows[i][c] >= 5)) {
                        isAllReverse = false;
                        break;
                    }
                }
                if (isAllReverse) {
                    highlightR1 = i;
                    outputP.value = i - rIdx;
                    break;
                }
            }
        }
        if (highlightR1 === -1) alert("Không tìm thấy hàng R1 ngược hoàn toàn!");
        refresh();
    }

    function render(x, k) {
        resultGrid.innerHTML = ''; 
        if (!currentNumbers.length) return;

        const mode = localStorage.getItem('userMode') || 'stepK';
        resultGrid.style.gridTemplateColumns = `var(--stt-width) var(--spacer-width) repeat(${x}, var(--cell-size))`;
        
        const rows = [];
        for (let i = 0; i < currentNumbers.length; i += x) rows.push(currentNumbers.slice(i, i + x));
        const marks = Array.from({length: rows.length}, () => Array(x).fill(''));
        let count = 0;

        if (mode === 'stepK') {
            for (let r = 0; r < rows.length - k; r++) {
                if (rows[r].length === x && rows[r+k].length === x) {
                    let isRev = true;
                    for (let c = 0; c < x; c++) { 
                        if ((rows[r][c] >= 5) === (rows[r+k][c] >= 5)) { isRev = false; break; } 
                    }
                    if (isRev) { 
                        count++; 
                        for (let c = 0; c < x; c++) marks[r][c] = marks[r+k][c] = 'blink'; 
                    }
                }
            }
        }

        const frag = document.createDocumentFragment();
        rows.forEach((row, rIdx) => {
            const stt = document.createElement('div'); 
            stt.className = 'row-stt'; 
            stt.textContent = rIdx + 1;
            frag.appendChild(stt); 
            frag.appendChild(document.createElement('div')); 

            row.forEach((n, cIdx) => {
                const box = document.createElement('div'); 
                box.className = 'number-box'; 
                box.textContent = n;
                
                // Quy tắc ưu tiên tô màu
                if (rIdx === highlightR) {
                    box.classList.add('row-r-base'); // Màu xanh da trời cho hàng R
                } else if (rIdx === highlightR1) {
                    box.classList.add('row-r1-found'); // Màu vàng cho hàng R1
                } else if (marks[rIdx][cIdx] === 'blink') {
                    box.classList.add('blink');
                } else if (mode === 'big' && n >= 5) {
                    box.classList.add('high');
                } else if (mode === 'small' && n < 5) {
                    box.classList.add('low');
                }
                frag.appendChild(box);
            });
            
            for (let i = row.length; i < x; i++) {
                const empty = document.createElement('div');
                empty.className = 'number-box';
                empty.style.opacity = "0.2";
                frag.appendChild(empty);
            }
        });
        resultGrid.appendChild(frag);
    }
</script>
