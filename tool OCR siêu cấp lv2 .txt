<script>
    const fileInput = document.getElementById('fileInput');
    const folderInput = document.getElementById('folderInput');
    const fileCountText = document.getElementById('file-count');
    const scanBtn = document.getElementById('scanBtn');
    const resultBody = document.getElementById('resultBody');
    const txtHorizontal = document.getElementById('txtHorizontal');
    const progressBar = document.getElementById('progressBar');
    const progressBox = document.getElementById('progressBox');
    const status = document.getElementById('status');
    const downloadBtn = document.getElementById('downloadBtn');
    const exportImgBtn = document.getElementById('exportImgBtn');
    const pdfBtn = document.getElementById('pdfBtn');
    const canvas = document.getElementById('canvas-helper');
    const ctx = canvas.getContext('2d');

    let allFiles = []; 
    let finalData = [];

    // Xử lý khi chọn File lẻ
    fileInput.addEventListener('change', handleFiles);
    // Xử lý khi chọn Thư mục
    folderInput.addEventListener('change', handleFiles);

    function handleFiles(e) {
        const newFiles = Array.from(e.target.files).filter(file => file.type.startsWith('image/'));
        if (newFiles.length > 0) {
            allFiles = allFiles.concat(newFiles);
            fileCountText.innerText = `Đã tích lũy: ${allFiles.length} ảnh`;
            scanBtn.disabled = false;
        }
        e.target.value = ""; // Reset để có thể chọn lại cùng thư mục nếu cần
    }

    scanBtn.addEventListener('click', async () => {
        if (allFiles.length === 0) return;

        scanBtn.disabled = true;
        progressBox.style.display = 'block';
        finalData = [];
        resultBody.innerHTML = "";

        const worker = await Tesseract.createWorker({
            cacheMethod: 'none',
            logger: m => { if(m.status === 'recognizing') console.log(m.progress); }
        });
        await worker.loadLanguage('eng');
        await worker.initialize('eng');

        for (let i = 0; i < allFiles.length; i++) {
            status.innerText = `Đang xử lý ${i + 1}/${allFiles.length}...`;
            progressBar.style.width = ((i + 1) / allFiles.length) * 100 + '%';

            try {
                const imgData = await preprocessImage(allFiles[i]);
                const { data: { text } } = await worker.recognize(imgData);
                parseText(text);
                URL.revokeObjectURL(imgData); 
            } catch (err) {
                console.error("Lỗi tại ảnh số " + i);
            }
        }

        await worker.terminate();
        finalData.sort((a, b) => b.kyXo.localeCompare(a.kyXo));
        renderTable(); 
        updateHorizontalData(); 

        status.innerHTML = `<b style="color:#2ecc71">Xong! Tổng: ${finalData.length} kỳ</b>`;
        downloadBtn.disabled = false;
        exportImgBtn.disabled = false;
        pdfBtn.disabled = false;
    });

    function preprocessImage(file) {
        return new Promise((resolve) => {
            const img = new Image();
            const objectUrl = URL.createObjectURL(file);
            img.onload = () => {
                canvas.width = img.width;
                canvas.height = img.height;
                ctx.filter = 'grayscale(1) contrast(1.2)';
                ctx.drawImage(img, 0, 0);
                const data = canvas.toDataURL('image/jpeg', 0.7);
                URL.revokeObjectURL(objectUrl); 
                resolve(data);
            };
            img.src = objectUrl;
        });
    }

    function parseText(text) {
        const lines = text.split('\n');
        lines.forEach(line => {
            const kyXoMatch = line.match(/\d{10,}/);
            if (kyXoMatch) {
                const kyXo = kyXoMatch[0];
                const sauKyXo = line.split(kyXo)[1] || "";
                const numMatch = sauKyXo.match(/\d/);
                if (numMatch) {
                    const conSo = parseInt(numMatch[0]);
                    let lonNho = (conSo >= 5) ? "Lớn" : "Nhỏ";
                    if(!finalData.some(item => item.kyXo === kyXo)) {
                        finalData.push({kyXo, conSo, lonNho});
                    }
                }
            }
        });
    }

    function renderTable() {
        let html = "";
        finalData.forEach(d => {
            const colorClass = (d.lonNho === "Lớn") ? "txt-lon" : "txt-nho";
            html += `<tr><td>${d.kyXo}</td><td class="${colorClass}">${d.conSo}</td><td class="${colorClass}">${d.lonNho}</td></tr>`;
        });
        resultBody.innerHTML = html;
    }

    function updateHorizontalData() {
        const sortedData = [...finalData].sort((a, b) => a.kyXo.localeCompare(b.kyXo));
        txtHorizontal.innerText = sortedData.map(item => item.conSo).join(" ");
    }

    downloadBtn.addEventListener('click', () => {
        const content = txtHorizontal.innerText;
        // Đếm số lượng chữ số thực tế (bỏ qua dấu cách)
        const digitCount = content.replace(/\s/g, '').length;
        
        const blob = new Blob([content], {type: 'text/plain;charset=utf-8'});
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        // Tên file dạng: vn168 - [số chữ số].txt
        a.download = `vn168 - ${digitCount}.txt`;
        document.body.appendChild(a);
        a.click();
        setTimeout(() => {
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
        }, 0);
    });

    exportImgBtn.addEventListener('click', () => {
        status.innerText = "Đang vẽ ảnh...";
        html2canvas(document.getElementById('tableCaptureArea'), {
            scale: 2,
            backgroundColor: "#000",
            useCORS: true
        }).then(canvas => {
            const link = document.createElement('a');
            link.download = `VN168_Export.png`;
            link.href = canvas.toDataURL("image/png");
            link.click();
            status.innerText = "✅ Xuất ảnh thành công!";
        });
    });

    pdfBtn.addEventListener('click', () => {
        window.print();
    });
</script>
