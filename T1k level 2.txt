<script>
    const fileInput = document.getElementById('fileInput');
    const rowLengthInput = document.getElementById('rowLength');
    const stepKInput = document.getElementById('stepK');
    const statusBar = document.getElementById('status-bar');
    const resultGrid = document.getElementById('result-grid');

    let currentNumbers = [];

    [rowLengthInput, stepKInput].forEach(el => {
        el.addEventListener('input', () => { if(currentNumbers.length > 0) refreshAll(); });
    });

    window.onload = function() {
        rowLengthInput.value = localStorage.getItem('userRowLength') || 2;
        stepKInput.value = localStorage.getItem('userStepK') || 1;
        const savedData = localStorage.getItem('userCurrentNumbers');
        if (savedData) {
            currentNumbers = JSON.parse(savedData);
            refreshAll();
        }
        updateHeaderGrid();
    };

    function clearMemory() {
        if(confirm("Xóa sạch dữ liệu và cài đặt?")) {
            localStorage.clear();
            location.reload();
        }
    }

    function showStatus(text, type) {
        statusBar.textContent = text;
        statusBar.className = 'status-' + type;
        statusBar.style.display = 'block';
    }

    function setMode(mode) {
        localStorage.setItem('userLastMode', mode);
        document.querySelectorAll('.main-btn').forEach(btn => btn.classList.remove('active-mode'));
        const activeBtn = document.getElementById('btn-' + mode);
        if(activeBtn) activeBtn.classList.add('active-mode');
        refreshAll();
    }

    function refreshAll() {
        const x = parseInt(rowLengthInput.value) || 2;
        const k = parseInt(stepKInput.value) || 1;
        localStorage.setItem('userRowLength', x);
        localStorage.setItem('userStepK', k);
        updateHeaderGrid();
        renderGrid(currentNumbers, x);
    }

    fileInput.addEventListener('change', function() {
        const reader = new FileReader();
        reader.onload = function(e) {
            const cleanText = e.target.result.replace(/[^0-9]/g, '');
            currentNumbers = cleanText.split('').map(Number);
            localStorage.setItem('userCurrentNumbers', JSON.stringify(currentNumbers));
            refreshAll();
        };
        reader.readAsText(this.files[0]);
    });

    function updateHeaderGrid() {
        const x = parseInt(rowLengthInput.value) || 2;
        const headerGrid = document.getElementById('header-grid');
        headerGrid.style.gridTemplateColumns = `var(--stt-width) var(--spacer-width) repeat(${x}, var(--cell-size))`;
        headerGrid.innerHTML = '<div style="width:var(--stt-width)"></div><div style="width:var(--spacer-width)"></div>';
        for (let i = 0; i < x; i++) {
            const box = document.createElement('div');
            box.className = `index-box idx-${i % 10}`;
            box.textContent = i % 10;
            headerGrid.appendChild(box);
        }
    }

    function renderGrid(data, columns) {
        resultGrid.innerHTML = '';
        if (data.length === 0) return;
        
        const mode = localStorage.getItem('userLastMode') || 'stepK';
        const k = parseInt(stepKInput.value) || 1;
        
        resultGrid.style.gridTemplateColumns = `var(--stt-width) var(--spacer-width) repeat(${columns}, var(--cell-size))`;
        const rows = [];
        for (let i = 0; i < data.length; i += columns) { rows.push(data.slice(i, i + columns)); }

        const markMap = Array.from({ length: rows.length }, () => Array(columns).fill(''));
        let matchCount = 0;

        if (mode === 'stepK') {
            for (let r = 0; r < rows.length - k; r++) {
                const rowU = rows[r], rowL = rows[r + k];
                // Chỉ kiểm tra khi cả 2 hàng đều đủ số cột
                if (rowU.length === columns && rowL.length === columns) {
                    let isStrictReverse = true;
                    for (let c = 0; c < columns; c++) {
                        // Logic: Lớn (>=5) và Nhỏ (<5)
                        const isUHigh = rowU[c] >= 5;
                        const isLHigh = rowL[c] >= 5;
                        // Nếu 2 ô cùng loại (cùng Lớn hoặc cùng Nhỏ) thì hàng này không "ngược" hoàn toàn
                        if (isUHigh === isLHigh) { 
                            isStrictReverse = false; 
                            break; 
                        }
                    }
                    if (isStrictReverse) {
                        matchCount++;
                        for (let c = 0; c < columns; c++) { 
                            markMap[r][c] = markMap[r+k][c] = 'blink'; 
                        }
                    }
                }
            }
            showStatus(matchCount > 0 ? `Tìm thấy ${matchCount} cặp ngược nhau!` : "Không có cặp nào ngược.", matchCount > 0 ? "success" : "info");
        }

        const fragment = document.createDocumentFragment();
        rows.forEach((rowData, rIndex) => {
            const stt = document.createElement('div'); stt.className = 'row-stt'; stt.textContent = rIndex + 1;
            fragment.appendChild(stt);
            fragment.appendChild(document.createElement('div'));

            rowData.forEach((num, cIndex) => {
                const box = document.createElement('div');
                box.className = 'number-box'; box.textContent = num;
                if (markMap[rIndex][cIndex] === 'blink') box.classList.add('blink');
                else if (mode === 'big' && num >= 5) box.classList.add('high');
                else if (mode === 'small' && num < 5) box.classList.add('low');
                fragment.appendChild(box);
            });
            for(let i = rowData.length; i < columns; i++) {
                fragment.appendChild(document.createElement('div')).className = 'number-box';
            }
        });
        resultGrid.appendChild(fragment);
    }
</script>
