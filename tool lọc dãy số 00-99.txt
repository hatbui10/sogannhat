<script>
    let rawData = localStorage.getItem('vn168_data') || "";
    let lastResults = []; // Lưu trữ kết quả hiện tại để tải về

    window.onload = () => {
        if(rawData) {
            document.getElementById('fileName').innerText = "Dữ liệu đã sẵn sàng";
            document.getElementById('fileName').style.color = "var(--neon-cyan)";
        }
        document.getElementById('searchInput').focus();
    };

    // Đọc file
    document.getElementById('fileInput').addEventListener('change', function(e) {
        const file = e.target.files[0];
        if (!file) return;
        const reader = new FileReader();
        reader.onload = (e) => {
            rawData = e.target.result.replace(/\s+/g, ''); // Nén chuỗi
            localStorage.setItem('vn168_data', rawData);
            document.getElementById('fileName').innerText = "Đã nạp: " + file.name;
            document.getElementById('fileName').style.color = "var(--neon-cyan)";
        };
        reader.readAsText(file);
    });

    function handleFilter() {
        const searchVal = document.getElementById('searchInput').value;
        const container = document.getElementById('resultContainer');
        
        if (!rawData) return alert("Bạn chưa tải file dữ liệu!");
        if (!searchVal) return;

        let resultsHTML = "";
        lastResults = []; // Reset bộ nhớ tải về
        let pos = rawData.indexOf(searchVal);

        while (pos !== -1) {
            let nextTen = rawData.substring(pos + searchVal.length, pos + searchVal.length + 10);
            
            if (nextTen.length > 0) {
                // Định dạng hiển thị
                let f = nextTen;
                if (nextTen.length === 10) {
                    f = `${nextTen.substring(0,3)} ${nextTen.substring(3,6)} ${nextTen.substring(6,10)}`;
                }
                
                lastResults.push(`${searchVal} ${f}`); // Lưu bản text để tải về
                resultsHTML += `
                    <div class="result-item">
                        <span class="match-val">${searchVal}</span>
                        <span class="sequence-val">${f}</span>
                    </div>`;
            }
            pos = rawData.indexOf(searchVal, pos + 1);
        }

        container.innerHTML = resultsHTML || '<div style="padding:20px; color:red; text-align:center;">Không tìm thấy!</div>';
        document.getElementById('searchInput').select();
    }

    // Chức năng tải file TXT
    function downloadResult() {
        if (lastResults.length === 0) return alert("Không có dữ liệu để tải!");
        
        const content = lastResults.join("\n");
        const blob = new Blob([content], { type: 'text/plain' });
        const url = window.URL.createObjectURL(blob);
        const a = document.createElement('a');
        
        const timestamp = new Date().toLocaleTimeString().replace(/:/g, '-');
        a.href = url;
        a.download = `KetQua_Loc_${timestamp}.txt`;
        a.click();
        window.URL.revokeObjectURL(url);
    }

    // Enter để lọc nhanh
    document.getElementById('searchInput').addEventListener('keypress', (e) => {
        if (e.key === 'Enter') handleFilter();
    });
</script>
