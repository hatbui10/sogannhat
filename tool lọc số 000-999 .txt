<script>
    let combinedData = ""; // Biến chứa dữ liệu cộng dồn
    let fileCount = 0;
    let results = { p1: [], p2: [], p3: [] };

    document.getElementById('fileInput').addEventListener('change', function(e) {
        const files = e.target.files;
        if (files.length === 0) return;

        fileCount += files.length;
        document.getElementById('fileStatus').innerText = `Đã cộng dồn: ${fileCount} file. Sẵn sàng xử lý...`;

        for (let file of files) {
            const reader = new FileReader();
            reader.onload = (event) => {
                // Cộng dồn nội dung vào biến chung
                combinedData += event.target.result.replace(/\s+/g, '');
            };
            reader.readAsText(file);
        }
    });

    function startProcessing() {
        if (!combinedData) return alert("Vui lòng chọn file trước!");
        
        results = { p1: [], p2: [], p3: [] };
        document.getElementById('fileStatus').innerText = "Đang xử lý dữ liệu tổng hợp...";

        // Quét từ 000-999
        for (let i = 0; i <= 999; i++) {
            let xxx = i.toString().padStart(3, '0');
            let pos = combinedData.indexOf(xxx);
            
            if (pos === -1) continue;

            let isP1 = false;
            let isP2 = false;

            // Kiểm tra TẤT CẢ các vị trí của xxx trong dữ liệu gộp
            let currentPos = pos;
            while (currentPos !== -1) {
                let nextSix = combinedData.substring(currentPos + 3, currentPos + 9);
                if (nextSix.length === 6) {
                    let nums = nextSix.split('').map(Number);
                    if (nums.every(n => n >= 5)) isP1 = true;
                    if (nums.every(n => n < 5)) isP2 = true;
                }
                currentPos = combinedData.indexOf(xxx, currentPos + 1);
            }

            // Phân loại ưu tiên
            if (isP1) results.p1.push(xxx);
            else if (isP2) results.p2.push(xxx);
            else results.p3.push(xxx);
        }

        updateUI();
        document.getElementById('fileStatus').innerText = `Hoàn tất xử lý ${fileCount} file!`;
    }

    function updateUI() {
        document.getElementById('countP1').innerText = results.p1.length;
        document.getElementById('countP2').innerText = results.p2.length;
        document.getElementById('countP3').innerText = results.p3.length;

        document.getElementById('listP1').innerText = results.p1.join('\n');
        document.getElementById('listP2').innerText = results.p2.join('\n');
        document.getElementById('listP3').innerText = results.p3.join('\n');
    }

    function resetAll() {
        combinedData = "";
        fileCount = 0;
        results = { p1: [], p2: [], p3: [] };
        document.getElementById('fileStatus').innerText = "Đã làm mới bộ nhớ.";
        updateUI();
    }

    function downloadPart(name) {
        if (results[name].length === 0) return;
        const blob = new Blob([results[name].join('\n')], { type: 'text/plain' });
        const a = document.createElement('a');
        a.href = URL.createObjectURL(blob);
        a.download = `VN168_GOP_${name.toUpperCase()}.txt`;
        a.click();
    }
</script>
