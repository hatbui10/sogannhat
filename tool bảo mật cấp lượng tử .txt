<script>
    let selShapes = [], selIcons = [], selSpecs = [];
    let failCount = 0, cycle = 0;

    const shapeChaos = {0:'A#7zL', 1:'pQ!2n', 2:'b*3Ty', 3:'hZ^8j', 4:'qN~0g', 5:'rT$2v', 6:'xM*7z', 7:'nJ!5h', 8:'mP(0e', 9:'kL#6u'};
    const iconChaos = {
        0:'Sm!Le7@9kR', 1:'HeArT#2n&5', 2:'FiRe*3TyM1', 3:'StAr^8j4(E', 4:'LiKe~0gf+6',
        5:'HaHa$2vB#9', 6:'AnGrY*7z@1', 7:'CrY!5h^4sD', 8:'SuPrI(0e&3', 9:'KiSs#6u+8j',
        10:'GuN@1n$9mX', 11:'BoMb#4k*7p', 12:'ClOuD^2s&0', 13:'MoOn!8m#5v', 14:'SuN(3z+6fR',
        15:'BiKe$9k^1j', 16:'CaR*0n#4mQ', 17:'PlAnE&7h@2', 18:'ShIp!5v^8r', 19:'KeY#1m@6nJ'
    };

    const specs = ['!', '@', '#', '$', '%', '^', '&', '*', '(', ')', '-', '+', '=', '{', '}', '[', ']', ';', ':', '?', '/', '>', '<', '~', '|'];
    
    const shapeSVGs = [
        '<circle cx="12" cy="12" r="9"/>','<rect x="4" y="4" width="16" height="16"/>','<path d="M12 2 L22 20 L2 20 Z"/>','<path d="M12 2 L15 9 L22 9 L17 14 L19 21 L12 17 L5 21 L7 14 L2 9 L9 9 Z"/>','<ellipse cx="12" cy="12" rx="9" ry="5"/>',
        '<path d="M3 12 h18 M12 3 v18"/>','<path d="M5 5 L19 19 M19 5 L5 19"/>','<circle cx="12" cy="12" r="4"/>','<path d="M12 21 a9 9 0 1 0 0 -18 v18"/>','<rect x="8" y="2" width="8" height="20" rx="2"/>'
    ];

    const socialEmojis = ['😊','❤️','🔥','⭐','👍','😆','😡','😭','😲','😘','🔫','💣','☁️','🌙','☀️','🚲','🚗','✈️','🚢','🔑'];

    // Khởi tạo lưới ký tự đặc biệt
    const spGrid = document.getElementById('specGrid');
    specs.forEach(s => {
        const d = document.createElement('div'); d.className = 'item-select'; d.innerText = s;
        d.onclick = () => { if(selSpecs.includes(s)) selSpecs = selSpecs.filter(x => x !== s); else selSpecs.push(s); d.classList.toggle('active'); };
        spGrid.appendChild(d);
    });

    // Khởi tạo lưới hình vẽ
    const sGrid = document.getElementById('shapeGrid');
    shapeSVGs.forEach((svg, i) => {
        const d = document.createElement('div'); d.className = 'item-select'; d.innerHTML = `<svg viewBox="0 0 24 24">${svg}</svg>`;
        d.onclick = () => { if(selShapes.includes(i)) selShapes = selShapes.filter(x => x !== i); else selShapes.push(i); d.classList.toggle('active'); };
        sGrid.appendChild(d);
    });

    // Khởi tạo lưới icon
    const iGrid = document.getElementById('iconGrid');
    socialEmojis.forEach((emoji, i) => {
        const d = document.createElement('div'); d.className = 'item-select'; d.innerText = emoji;
        d.onclick = () => { if(selIcons.includes(i)) selIcons = selIcons.filter(x => x !== i); else selIcons.push(i); d.classList.toggle('active'); };
        iGrid.appendChild(d);
    });

    document.getElementById('c').oninput = function() { document.getElementById('hexVal').innerText = this.value.toUpperCase(); document.getElementById('hexVal').style.color = this.value; };

    const penaltyTiers = [300, 600, 1800, 3600, 7200, 14400, 86400];

    function startTimer(seconds) {
        const bE = document.getElementById('btnEnc'); const bD = document.getElementById('btnDec'); const t = document.getElementById('timer');
        bE.disabled = bD.disabled = true; t.style.display = 'block';
        let remaining = seconds;
        const interval = setInterval(() => {
            let h = Math.floor(remaining / 3600); let m = Math.floor((remaining % 3600) / 60); let s = remaining % 60;
            t.innerText = `HỆ THỐNG ĐANG KHÓA: ${h > 0 ? h+':' : ''}${m < 10 ? '0'+m : m}:${s < 10 ? '0'+s : s}`;
            if (--remaining < 0) { clearInterval(interval); bE.disabled = bD.disabled = false; t.style.display = 'none'; }
        }, 1000);
    }

    function triggerAction(mode) {
        document.getElementById('loaderText').innerText = mode === 'enc' ? "HỆ THỐNG ĐANG MÃ HÓA..." : "HỆ THỐNG ĐANG GIẢI MÃ...";
        document.getElementById('loader').style.display = 'flex';
        setTimeout(() => { run(mode); }, 100);
    }

    async function run(m) {
        const loader = document.getElementById('loader');
        const file = document.getElementById('f').files[0]; const ans = document.getElementById('a').value;
        const num = document.getElementById('n').value; const hex = document.getElementById('c').value;
        
        if (!file || !ans || num.length < 12 || selShapes.length === 0 || selIcons.length === 0 || selSpecs.length === 0) {
            alert("Vui lòng hoàn thành đủ thông tin (Ký tự, Hình vẽ và Icon)!"); 
            loader.style.display = 'none';
            return;
        }

        const specStr = selSpecs.sort().join('');
        const shapeStr = selShapes.sort((a,b)=>a-b).map(id => shapeChaos[id]).join('');
        const iconStr = selIcons.sort((a,b)=>a-b).map(id => iconChaos[id]).join('');
        const raw = ans + num + specStr + shapeStr + iconStr + hex;
        const salt = CryptoJS.SHA512(hex + specStr.length + shapeStr.length + iconStr.length).toString();

        const r = new FileReader();
        r.onload = function(e) {
            try {
                const k = CryptoJS.PBKDF2(raw, salt, { keySize: 512/32, iterations: 100000, hasher: CryptoJS.algo.SHA512 }).toString();
                if (m === 'enc') {
                    const encrypted = CryptoJS.AES.encrypt(e.target.result, k).toString();
                    const a = document.createElement('a'); a.href = URL.createObjectURL(new Blob(["KSS_FULL|" + encrypted])); a.download = `SECURE_V7.txt`; a.click();
                    loader.style.display = 'none';
                    location.reload();
                } else {
                    let data = e.target.result; if(data.startsWith("KSS_FULL|")) data = data.split("|")[1];
                    const dec = CryptoJS.AES.decrypt(data, k).toString(CryptoJS.enc.Utf8);
                    if (!dec) throw new Error();
                    document.getElementById('o').value = dec;
                    failCount = 0; cycle = 0;
                    loader.style.display = 'none';
                }
            } catch (err) {
                loader.style.display = 'none';
                failCount++;
                if (failCount % 5 === 0) { let p = penaltyTiers[cycle] || 86400; startTimer(p); cycle++; }
                else { alert(`Sai mật mã! Bạn còn ${5 - (failCount % 5)} lần thử.`); }
            }
        };
        r.readAsText(file);
    }
</script>
