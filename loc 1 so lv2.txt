<script>
const ni = document.getElementById('n-input');
const n1 = document.getElementById('n1-box');
const ls = document.getElementById('list');

let digits = JSON.parse(localStorage.getItem('dg')) || [];
let cur = localStorage.getItem('cn') || "";
let hist = [];
let cells = [];

/* RENDER (đã tối ưu) */
function render() {
    ls.innerHTML = digits.map(d => `<div class="c" data-v="${d}">${d}</div>`).join('');
    cells = ls.children; // cache DOM
}

/* TÔ N (không query lại DOM) */
function colorByN() {
    for (let el of cells) el.classList.remove('match');
    if (!cur) return;

    for (let el of cells) {
        if (el.dataset.v == cur) el.classList.add('match');
    }
}

/* LOGIC Ab (đã tối ưu) */
function findPattern() {
    for (let el of cells) el.classList.remove('match');

    if (!cur || digits.length < 2) return;

    let pattern = [];
    let pos = [];

    for (let i = 0; i < digits.length - 1; i++) {
        if (digits[i] == cur) {
            let C = digits[i+1];
            pattern.push(C >= 5 ? "A" : "b");
            pos.push(i);
        }
    }

    for (let i = 0; i <= pattern.length - 6; i++) {
        if (
            pattern[i] === "A" &&
            pattern[i+1] === "b" &&
            pattern[i+2] === "A" &&
            pattern[i+3] === "b" &&
            pattern[i+4] === "A" &&
            pattern[i+5] === "b"
        ) {
            for (let j = 0; j < 6; j++) {
                let idx = pos[i + j];
                if (cells[idx]) cells[idx].classList.add('match');
                if (cells[idx+1]) cells[idx+1].classList.add('match');
            }
        }
    }
}

/* LOAD FILE */
document.getElementById('f').onchange = e => {
    const r = new FileReader();
    r.onload = () => {
        digits = r.result.replace(/\D/g,'').split('');
        localStorage.setItem('dg', JSON.stringify(digits));
        render();
    };
    r.readAsText(e.target.files[0]);
};

/* INPUT */
ni.oninput = () => {
    if (ni.value !== "") {
        hist.push(cur);
        cur = ni.value;
        n1.innerText = cur;
        localStorage.setItem('cn', cur);
        ni.value = "";
        colorByN();
    }
};

function undo() {
    if (hist.length) {
        cur = hist.pop();
        n1.innerText = cur || "-";
        colorByN();
    }
}

function clearAll() {
    if (confirm("Xoá sạch dữ liệu?")) {
        localStorage.clear();
        digits = [];
        cur = "";
        ls.innerHTML = "";
        n1.innerText = "-";
    }
}

window.onload = () => {
    n1.innerText = cur || "-";
    render();
};
