    function updateBadges(containerId, targetArray, chaosData) {
        const items = document.getElementById(containerId).children;
        for (let i = 0; i < items.length; i++) {
            let val;
            if (chaosData) { val = chaosData[i]; } 
            else { val = items[i].textContent.replace(/[0-9]/g, '').trim(); }
            const orderIdx = targetArray.indexOf(val);
            const badge = items[i].querySelector('.badge');
            if (orderIdx > -1) {

        const grid = document.getElementById(containerId);

                }

        };
