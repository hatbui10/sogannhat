<script>
    const STORAGE_KEY = 'MATRIX_FINAL_N345_B4';
    const STATUS_KEY = 'MATRIX_STATUS_N345_B4';
    let currentResults = [];

    document.addEventListener('DOMContentLoaded', () => {
        const savedData = localStorage.getItem(STORAGE_KEY);
        const savedStatus = localStorage.getItem(STATUS_KEY);
        if (savedData) {
            document.getElementById('resultArea').innerHTML = savedData;
            document.getElementById('status').innerText = savedStatus || "Đã khôi phục dữ liệu.";
            updateCurrentResultsFromUI();
        }
    });

    // Logic kiểm tra BỆT 4
    function hasFourInARow(str) {
        if (str.length < 4) return false;
        let countA = 0, countB = 0;
        for (let i = 0; i < str.length; i++) {
            const val = parseInt(str[i]);
            if (val >= 5) { countA++; countB = 0; } 
            else { countB++; countA = 0; }
            if (countA >= 4 || countB >= 4) return true;
        }
        return false;
    }

    function handleMode(n) {
        const fileInput = document.getElementById('fileUpload');
        const status = document.getElementById('status');
        if (!fileInput.files[0]) return alert("Vui lòng chọn file trước!");

        const reader = new FileReader();
        reader.onload = function(e) {
            const allDigits = e.target.result.replace(/\D/g, ''); 
            const registry = {};

            for (let i = 0; i <= allDigits.length - n; i++) {
                const currentChunk = allDigits.substring(i, i + n);
                const nextSix = allDigits.substring(i + n, i + n + 6);

                if (!(currentChunk in registry)) {
                    registry[currentChunk] = { isAlwaysSafe: true };
                }

                // Nếu bất kỳ lần xuất hiện nào dính bệt 4 -> Loại vĩnh viễn (false)
                if (nextSix.length >= 4) {
                    if (hasFourInARow(nextSix)) {
                        registry[currentChunk].isAlwaysSafe = false;
                    }
                }
            }
            
            // Chỉ lấy những cụm số LUÔN LUÔN an toàn trong suốt file
            currentResults = Object.keys(registry)
                .filter(num => registry[num].isAlwaysSafe)
                .sort();

            const statusText = `Đã lọc tuyệt đối n=${n} (Né bệt 4). Tìm thấy ${currentResults.length} số.`;
            status.innerText = statusText;
            
            renderResults(currentResults);
            saveToLocal(statusText);
        };
        reader.readAsText(fileInput.files[0]);
    }

    function renderResults(arr) {
        const resultArea = document.getElementById('resultArea');
        resultArea.innerHTML = arr.map(num => `<div class="num-row">${num}</div>`).join('');
    }

    function saveToLocal(statusText) {
        localStorage.setItem(STORAGE_KEY, document.getElementById('resultArea').innerHTML);
        localStorage.setItem(STATUS_KEY, statusText);
    }

    function updateCurrentResultsFromUI() {
        const rows = document.querySelectorAll('.num-row');
        currentResults = Array.from(rows).map(row => row.innerText);
    }

    function downloadResult() {
        if (currentResults.length === 0) return alert("Không có dữ liệu!");
        const content = currentResults.join('\n');
        const blob = new Blob([content], { type: 'text/plain' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = `absolute_safe_n${currentResults[0].length}_b4.txt`;
        document.body.appendChild(a);
        a.click();
        document.body.removeChild(a);
    }

    function clearData() {
        if(confirm("Xóa sạch dữ liệu lưu trữ?")) {
            localStorage.removeItem(STORAGE_KEY);
            localStorage.removeItem(STATUS_KEY);
            document.getElementById('resultArea').innerHTML = '';
            document.getElementById('status').innerText = "Đã dọn dẹp.";
            currentResults = [];
        }
    }
</script>
